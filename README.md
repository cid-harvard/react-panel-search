```bash
npm install --save react-panel-search
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-panel-search'
import 'react-panel-search/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
